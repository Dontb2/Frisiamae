function reverseNumber() {
// Get the number from the input field
const number = document.getElementById('numberInput').value;
let num = Math.abs(parseInt(number)); // Ensure the number is positive
let reversed = 0;

// Reverse the number using a do-while loop
do {
reversed = reversed * 10 + (num % 10); // Extract the last digit and add it to reversed
num = Math.floor(num / 10); // Remove the last digit
} while (num > 0);
// Display the result
const resultDiv = document.getElementById('result');
if (number) {
const reversedNumber = parseInt(number) < 0 ? -reversed : reversed; // Handle negative numbers
resultDiv.innerHTML =

The reverse of ${number} is: ${reversedNumber}

;
} else {
resultDiv.innerHTML =

Please enter a valid number.

;
}
}
