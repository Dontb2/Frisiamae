script>
function btn() {
const day = parseInt(document.getElementById("number").value);
let resultText = "";
switch (day) {
case 1:
resultText = "Sunday";
break;
case 2:
resultText = "Monday";
break;
case 3:
resultText = "Tuesday";
break;
case 4:
resultText = "Wednesday";
break;
case 5:
resultText = "Thursday";
break;
case 6:
resultText = "Friday";
break;
case 7:
resultText = "Saturday";
break;
default:
resultText = "Invalid Day";
}
document.getElementById("result").innerHTML = resultText;
}
