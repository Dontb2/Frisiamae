function categorizeFood() {

const foodItem = document.getElementById('foodInput').value.trim().toLowerCase();
const goFoods = ['rice', 'bread', 'pasta', 'potatoes', 'corn', 'vegetables'];
const growFoods = ['meat', 'fish', 'chicken', 'eggs', 'beans', 'tofu', 'cheese'];
const glowFoods = ['carrot', 'spinach', 'tomato', 'cucumber', 'apples', 'oranges', 'berries', 'banana'];
const resultDiv = document.getElementById('result');
if (goFoods.includes(foodItem)) {
resultDiv.innerHTML = "Go Food:";
} else if (growFoods.includes(foodItem)) {
resultDiv.innerHTML = "Grow Food:";
} else if (glowFoods.includes(foodItem)) {
resultDiv.innerHTML = "Glow Food:";
} else {
resultDiv.innerHTML = "no result";
}
}
